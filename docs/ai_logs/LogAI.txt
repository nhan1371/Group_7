CREATE PROCEDURE generate_users_vn()
BEGIN
    DECLARE i INT DEFAULT 1;

    WHILE i <= 10000 DO
        INSERT INTO users
        (username, full_name, email, phone, password, address, city, loyalty_points, is_active, created_at)
        VALUES (
            CONCAT('user', i),

            CONCAT(
                (SELECT name FROM vn_last_name ORDER BY RAND() LIMIT 1),
                ' ',
                (SELECT name FROM vn_first_name ORDER BY RAND() LIMIT 1)
            ),

            CONCAT('user', i, '@gmail.com'),

            CONCAT(
                '0',
                ELT(FLOOR(1 + RAND() * 4), '91', '93', '97', '98'),
                LPAD(FLOOR(RAND() * 10000000), 7, '0')
            ),

            'hashed_password_here',

            CONCAT(
                FLOOR(1 + RAND() * 300),
                ' ',
                ELT(FLOOR(1 + RAND() * 4),
                    'Le Loi',
                    'Nguyen Trai',
                    'Tran Hung Dao',
                    'Vo Van Tan'
                )
            ),

            (SELECT name FROM vn_city ORDER BY RAND() LIMIT 1),

            FLOOR(RAND() * 5000),

            IF(RAND() < 0.9, 1, 0),

            DATE_SUB(NOW(), INTERVAL FLOOR(RAND() * 720) DAY)
        );

        SET i = i + 1;
    END WHILE;
END$$
